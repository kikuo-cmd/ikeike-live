// 1. 通信設定を強化（強制的に通り道を探す設定） const client = AgoraRTC.createClient({ mode: "live", codec: "vp8" }); // ネットワークが厳しい環境用のリトライ設定 AgoraRTC.setLogLevel(1); // ログを詳細に出す let localTracks = { videoTrack: null, audioTrack: null }; let remoteUser = null; const statusText = document.getElementById('status-bar'); const playOverlay = document.getElementById('play-overlay'); client.on("user-published", async (user, mediaType) => { try { statusText.innerText = "信号を検知しました..."; await client.subscribe(user, mediaType); remoteUser = user; if (mediaType === "video") { statusText.innerText = "再生準備完了！"; playOverlay.style.display = 'flex'; } if (mediaType === "audio") { user.audioTrack.play(); } } catch (e) { statusText.innerText = "受信エラー: " + e.message; } }); async function startStream(role) { const appId = document.getElementById('appid').value.trim(); const channel = document.getElementById('channel').value.trim(); if (!appId) return alert("App IDを入力してください"); try { statusText.innerText = "サーバーに接続を試みています..."; await client.setClientRole(role); // トークンなし(null)で参加 await client.join(appId, channel, null, null); statusText.innerText = "サーバー接続成功。信号を待機中..."; if (role === 'host') { statusText.innerText = "画面共有をリクエスト中..."; // Chromebook向け：解像度を少し下げて安定させる localTracks.videoTrack = await AgoraRTC.createScreenVideoTrack({ encoderConfig: { width: 640, height: 480, frameRate: 15 } }); localTracks.audioTrack = await AgoraRTC.createMicrophoneAudioTrack(); document.getElementById('remote-video').style.display = 'none'; localTracks.videoTrack.play("local-video"); await client.publish(Object.values(localTracks)); statusText.innerText = "🔴 配信中！相手に同じルーム名を伝えてください"; } } catch (err) { console.error(err); // 画面にエラーを直接出す statusText.innerText = "❌ 接続失敗: " + err.code; if(err.code === "CAN_NOT_GET_GATEWAY_SERVER") { statusText.innerText = "❌ ネットワーク制限（ファイアウォール）で遮断されました"; } } }
